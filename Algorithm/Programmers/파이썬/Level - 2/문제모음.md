1. capitalize() 앞글자만 대문자, 나머지 소문자

```

def solution(s):
    answer = ''
    
    for i in s.split(" "):
        if answer == "":
            answer += i.capitalize()
        else:
            answer += " " + i.capitalize()
     
    return answer

```

```
def solution(arr1, arr2):
    a, b = len(arr1), len(arr2[0])
    answer = [[0] * b for _ in range(a)]

    for i in range(a) :
        for j in range(b) :
            for z in range(len(arr2)) :
                answer[i][j] += arr1[i][z] * arr2[z][j]

    return answer
```

# 숫자의 표현

```

def solution(n):
    answer = 1
    
    for i in range(1, 1 + n // 2):
        plus = 0
        current = i
        while plus < n:
            plus += current
            current += 1
        
        if plus == n:
            answer+=1
        
    
    return answer
```

# 다음 큰 숫자

1. bin => n을 binary그리고 .count('1') => 1의 개수

```

def solution(n):
    
    count = bin(n).count('1')
    
    num_cnt = 0
    
    while num_cnt != count:
        n += 1
        num_cnt = bin(n).count('1')
    
    
    return int(bin(n), 2)


```

```
def solution(s):
    s = list(s)
    cnt = 0
    brackets = ['(',  '', ')']

    for i in s :
        cnt += brackets.index(i)-1
        if cnt > 0 :
            return False

    return True if cnt == 0 else False


def solution(s):
    
    stack = []
    
    for i in s:
        if i == '(':
            stack.append(i)
        else:
            if not stack:
                return False
            else:
                stack.pop()
    
    
    return True if not stack else False
```
